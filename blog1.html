Project 1: Azure Service Principal Authentication

Secure authentication for Azure resources using Service Principals

Overview

This project demonstrates how to authenticate Terraform with Azure using Service Principals with Client Secrets, providing secure access to Azure resources without interactive login.

Key Features

  • Service Principal creation and configuration
  • Client Secret authentication
  • Role-based access control (RBAC)
  • Secure credential management

File Structure

├── main.tf
├── variables.tf
├── outputs.tf
├── providers.tf
├── terraform.tfvars.example
├── cloud-init.yaml
└── README.md

Implementation Code

providers.tf

providers.tf 
terraform {
  required_version = ">= 1.0"
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0"
    }
  }
}

provider "azurerm" {
  features {}

  subscription_id = var.subscription_id
  client_id       = var.client_id
  client_secret   = var.client_secret
  tenant_id       = var.tenant_id
}

variables.tf

variables.tf 
variable "subscription_id" {
  description = "Azure Subscription ID"
  type        = string
  sensitive   = true
}

variable "client_id" {
  description = "Azure Service Principal App ID"
  type        = string
  sensitive   = true
}

variable "client_secret" {
  description = "Azure Service Principal Client Secret"
  type        = string
  sensitive   = true
}

variable "tenant_id" {
  description = "Azure Tenant ID"
  type        = string
  sensitive   = true
}

variable "resource_group_name" {
  description = "Name of the resource group"
  type        = string
  default     = "tf-service-principal-rg"
}

variable "location" {
  description = "Azure region"
  type        = string
  default     = "East US"
}

main.tf

main.tf 
# Create Resource Group
resource "azurerm_resource_group" "main" {
  name     = var.resource_group_name
  location = var.location

  tags = {
    environment = "demo"
    project     = "service-principal-auth"
  }
}

# Create Virtual Network
resource "azurerm_virtual_network" "main" {
  name                = "vnet-main"
  address_space       = ["10.0.0.0/16"]
  location            = azurerm_resource_group.main.location
  resource_group_name = azurerm_resource_group.main.name

  tags = {
    environment = "demo"
  }
}

# Create Subnet
resource "azurerm_subnet" "internal" {
  name                 = "subnet-internal"
  resource_group_name  = azurerm_resource_group.main.name
  virtual_network_name = azurerm_virtual_network.main.name
  address_prefixes     = ["10.0.1.0/24"]
}

# Create Network Security Group
resource "azurerm_network_security_group" "main" {
  name                = "nsg-main"
  location            = azurerm_resource_group.main.location
  resource_group_name = azurerm_resource_group.main.name

  security_rule {
    name                       = "SSH"
    priority                   = 1001
    direction                  = "Inbound"
    access                     = "Allow"
    protocol                   = "Tcp"
    source_port_range          = "*"
    destination_port_range     = "22"
    source_address_prefix      = "*"
    destination_address_prefix = "*"
  }

  tags = {
    environment = "demo"
  }
}

Deployment Steps

Step 1: Create Service Principal

az ad sp create-for-rbac --name "terraform-sp" --role Contributor

Step 2: Configure Environment Variables

Environment Setup 
export ARM_SUBSCRIPTION_ID="your-subscription-id"
export ARM_CLIENT_ID="your-client-id"
export ARM_CLIENT_SECRET="your-client-secret"
export ARM_TENANT_ID="your-tenant-id"

Step 3: Terraform Deployment

terraform init - Initialize Terraform
terraform plan - Review deployment plan
terraform apply - Apply configuration

Project 2: Multi-Cloud + Multi-Region Deployment

Deploy infrastructure across AWS and Azure in multiple regions

Overview

This advanced project demonstrates how to deploy infrastructure across multiple cloud providers (AWS and Azure) and multiple regions, showcasing Terraform's multi-cloud capabilities.

Key Features

  • Multi-provider configuration (AWS + Azure)
  • Cross-region deployments
  • Modular architecture
  • Shared networking configuration

Implementation Code

providers.tf

providers.tf 
terraform {
  required_version = ">= 1.0"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0"
    }
  }
}

# AWS Provider - Multiple Regions
provider "aws" {
  region = var.aws_primary_region
  alias  = "primary"
}

provider "aws" {
  region = var.aws_secondary_region
  alias  = "secondary"
}

# Azure Provider - Multiple Regions
provider "azurerm" {
  features {}
  subscription_id = var.azure_subscription_id
  client_id       = var.azure_client_id
  client_secret   = var.azure_client_secret
  tenant_id       = var.azure_tenant_id
  alias           = "primary"
}

provider "azurerm" {

  client_id       = var.azure_client_id

main.tf

main.tf 
# AWS Primary Region Resources
module "aws_primary" {
  source = "./modules/aws-ec2"

  providers = {
    aws = aws.primary
  }

  environment    = var.environment
  instance_type  = var.aws_instance_type
  instance_count = var.aws_instance_count
  vpc_cidr       = var.aws_vpc_cidr_primary
  region         = var.aws_primary_region
}

# AWS Secondary Region Resources
module "aws_secondary" {
  source = "./modules/aws-ec2"

  providers = {
    aws = aws.secondary
  }

  environment    = var.environment
  instance_type  = var.aws_instance_type
  instance_count = var.aws_instance_count
  vpc_cidr       = var.aws_vpc_cidr_secondary
  region         = var.aws_secondary_region
}

# Azure Primary Region Resources
module "azure_primary" {
  source = "./modules/azure-vm"

  providers = {
    azurerm = azurerm.primary
  }

  environment          = var.environment
  vm_size              = var.azure_vm_size
  instance_count       = var.azure_instance_count
  resource_group_name  = "${var.environment}-rg-primary"
  location            = var.azure_primary_region
  vnet_address_space  = var.azure_vnet_cidr_primary
}

# Azure Secondary Region Resources
module "azure_secondary" {
  source = "./modules/azure-vm"

  providers = {
    azurerm = azurerm.secondary
  }

  environment          = var.environment
  vm_size              = var.azure_vm_size
  instance_count       = var.azure_instance_count
  resource_group_name  = "${var.environment}-rg-secondary"
  location            = var.azure_secondary_region
  vnet_address_space  = var.azure_vnet_cidr_secondary
}

# Shared Resources (Load Balancers, DNS, etc.)
module "shared" {
  source = "./modules/shared"

  aws_primary_instances   = module.aws_primary.instance_ips
  aws_secondary_instances = module.aws_secondary.instance_ips
  azure_primary_instances = module.azure_primary.instance_ips
  azure_secondary_instances = module.azure_secondary.instance_ips

  domain_name = var.domain_name
}
